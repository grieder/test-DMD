146 void print(FILE* fp, const char *what, int dimension = 3, const char *format = " %lg", bool brackets = true, bool drop_tiny_off = false, double thr = 1e-2) const{
147 fprintf(fp, what);
148 double diag_min = abs(m[0][0]), off_max = 0;
149 for (int i = 0; i < dimension; i++)
150 for (int j = 0; j < dimension; j++){
151 if (i == j && abs(m[i][i]) < diag_min) { diag_min = abs(m[i][i]); }
152 if (i != j && abs(m[i][j]) > off_max) { off_max = abs(m[i][j]); }
153 }
154 if (drop_tiny_off && off_max < thr * diag_min){
155 for (int i = 0; i < dimension; i++){ fprintf(fp, format, m[i][i]); }
156 fprintf(fp, "\n");
157 }
158 else{
159 fprintf(fp, "\n");
160 for (int i = 0; i < dimension; i++){
161 if (brackets) fprintf(fp, "[ ");
162 for (int j = 0; j < dimension; j++) fprintf(fp, format, m[i][j]);
163 if (brackets) fprintf(fp, " ]\n"); else fprintf(fp, "\n");

289__hostanddev__ matrix3<> rotation(double theta, int axis)
290{ double s, c; sincos(theta, &s, &c);
291 switch(axis)
292 { case 0:
293 return matrix3<>(

311struct SpaceGroupOp
312{ matrix3<int> rot;

314
315 SpaceGroupOp(matrix3<int> rot = matrix3<int>(1,1,1), vector3<> a = vector3<>(0,0,0)) : rot(rot), a(a) {}
316
317 vector3<> applyReal(const vector3<>& r) const { return rot * r + a; }
318 vector3<> applyRecip(const vector3<>& k) const { return k * rot; }
319
321 SpaceGroupOp operator*(const SpaceGroupOp& op) const
322 { return SpaceGroupOp(rot*op.rot, rot*op.a + a);
323 }
324
326 SpaceGroupOp inv() const
327 { SpaceGroupOp result;
328 result.rot = det(rot) * adjugate(rot); //using det(rot) = +/-1 for symmetry matrix
329 result.a = -(result.rot * a);
330 return result;
